myArraySlice.js (reducer file)


import { createSlice } from "@reduxjs/toolkit"


const initialState = {

myArray: [ ],

};


const myArraySlice = createSlice({

name: "myArray",

initialState,

reducers: {


saveToMyArray: (state, action) => {

state.myArray.push(action.payload);

},


deletFromMyArray: (state, action) => {

let tempArray = [ ...state.myArray ];


const index = tempArray.findIndex((i) => action.payload === i.id);

if (index >= 0) {

tempArray.splice(index, 1);

} else {

console.warn("no object in array");

}

state.myArray = [ ...tempArray ];

}

});


action.payload it is the value passed on


export const { saveToMyArray, deletFromMyArray } = myArraySlice.actions


export const selectMyArray = (state) => state.myArray.myArray


export default myArraySlice.reducer
